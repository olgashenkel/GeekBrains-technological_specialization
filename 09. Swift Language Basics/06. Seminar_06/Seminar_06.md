```
// ************* 1. Работа с адресами массивов (Copy-on-Write) *************
// Вспомогательная функция для получения адреса памяти массива
func getAddress<T>(_ array: [T]) -> String {
    return array.withUnsafeBufferPointer { String(describing: $0.baseAddress) }
}

// 1. Создаем исходный массив
var a = [Tea(type: .black, cost: 50.0), Tea(type: .green, cost: 60.0)]
var b = a

print("До изменения:")
print("Адрес массива 'a': \(getAddress(a))")
print("Адрес массива 'b': \(getAddress(b))")
// ОБЪЯСНЕНИЕ: Адреса одинаковые! Swift экономит память и не копирует данные, 
// пока мы просто присваиваем одну переменную другой (механизм Copy-on-Write).

// Изменяем массив 'b'
b.append(Tea(type: .fruit, cost: 70.0))

print("\nПосле изменения 'b':")
print("Адрес массива 'a': \(getAddress(a))")
print("Адрес массива 'b': \(getAddress(b))")
// ОБЪЯСНЕНИЕ: Адреса стали РАЗНЫМИ. Как только мы попытались изменить массив 'b',
```

```
class MyClass {
    var test: String
    init(test: String) { self.test = test }
}

let c = MyClass(test: "Привет")
let d = c // Передается ссылка. 'c' и 'd' теперь указывают на один и тот же объект.

c.test = "Пока" // Меняем значение через переменную 'c'

print("Значение c.test: \(c.test)") // Выведет: "Пока"
print("Значение d.test: \(d.test)") // Выведет: "Пока"

// ОБЪЯСНЕНИЕ: Значение изменилось в обеих переменных, так как 'c' и 'd' ведут на один и тот же единственный экземпляр класса в памяти.
```

```
    var products: [Product] = []
```

```
// ************* 5. Проблема сильной циклической ссылки (Retain Cycle) *************

Ответ на вопрос:
Да, в такой связке есть критическая проблема. Она называется Retain Cycle (циклическая ссылка или сильная циклическая зависимость).

В чем суть проблемы:
По умолчанию все ссылки на классы в Swift являются сильными (strong) и увеличивают счетчик ссылок (ARC) объекта на 1. Магазин крепко держит Директора, а Директор крепко держит Магазин. Если эти объекты больше не будут нужны программе, они никогда не удалятся из оперативной памяти, что приведет к утечке памяти (Memory Leak).

Как решить:
Нужно разорвать этот цикл, сделав одну из ссылок слабой при помощи ключевого слова weak. Обычно ослабляют ссылку от подчиненного (или зависимого) объекта к главному. В данном случае логично сделать слабой ссылку на владельца внутри магазина. При weak ссылка автоматически станет опциональной (?), и если владелец удалится, ссылка внутри магазина безопасно превратится в nil.

Реализация решения:
// Класс владельца
class Owner {
    let name: String
    // Сильная ссылка: Владелец владеет магазином
    var store: Store?
    
    init(name: String) {
        self.name = name
    }
}

// Пример использования без утечек памяти:
do {
    let boss = Owner(name: "Иван Иванович")
    let setupStore = Store()
    
    boss.store = setupStore
    setupStore.owner = boss // Переменная owner внутри Store помечена как weak (см. пункт 4)
} 
// При выходе из блока 'do' оба объекта успешно и безболезненно удалятся из памяти!
```
