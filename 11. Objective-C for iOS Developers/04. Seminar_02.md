#import <Foundation/Foundation.h>
#import <math.h>

// ==========================================
// 1. АБСТРАКТНЫЙ КЛАСС FIGURE
// ==========================================
@interface Figure : NSObject

- (double)area;
- (double)perimeter;
- (void)printInfo;

@end

@implementation Figure

- (double)area {
    // Симуляция абстрактного метода
    [NSException raise:NSInternalInconsistencyException
                format:@"Вы должны переопределить метод %s в подклассе", __PRETTY_FUNCTION__];
    return 0.0;
}

- (double)perimeter {
    [NSException raise:NSInternalInconsistencyException
                format:@"Вы должны переопределить метод %s в подклассе", __PRETTY_FUNCTION__];
    return 0.0;
}

- (void)printInfo {
    NSLog(@"Фигура: %s | Площадь: %.2f | Периметр: %.2f", 
          object_getClassName(self), [self area], [self perimeter]);
}

@end



// ==========================================
// 2. ПРОИЗВОДНЫЕ КЛАССЫ (ПОДКЛАССЫ)
// ==========================================

// --- RECTANGLE (Прямоугольник) ---
@interface Rectangle : Figure
@property (nonatomic, assign) double width;
@property (nonatomic, assign) double height;

- (instancetype)initWithWidth:(double)width height:(double)height;
@end

@implementation Rectangle
- (instancetype)initWithWidth:(double)width height:(double)height {
    self = [super init];
    if (self) {
        _width = width;
        _height = height;
    }
    return self;
}
- (double)area { return _width * _height; }
- (double)perimeter { return 2 * (_width + _height); }
@end


// --- CIRCLE (Круг) ---
@interface Circle : Figure
@property (nonatomic, assign) double radius;

- (instancetype)initWithRadius:(double)radius;
@end

@implementation Circle
- (instancetype)initWithRadius:(double)radius {
    self = [super init];
    if (self) { _radius = radius; }
    return self;
}
- (double)area { return M_PI * _radius * _radius; }
- (double)perimeter { return 2 * M_PI * _radius; }
@end


// --- TRIANGLE (Треугольник) ---
@interface Triangle : Figure
@property (nonatomic, assign) double sideA;
@property (nonatomic, assign) double sideB;
@property (nonatomic, assign) double sideC;

- (instancetype)initWithSideA:(double)a sideB:(double)b sideC:(double)c;
@end

@implementation Triangle
- (instancetype)initWithSideA:(double)a sideB:(double)b sideC:(double)c {
    self = [super init];
    if (self) {
        _sideA = a;
        _sideB = b;
        _sideC = c;
    }
    return self;
}
// Вычисление площади по формуле Герона
- (double)area {
    double p = [self perimeter] / 2.0;
    return sqrt(p * (p - _sideA) * (p - _sideB) * (p - _sideC));
}
- (double)perimeter { return _sideA + _sideB + _sideC; }
@end


// ==========================================
// 3. ГЛАВНАЯ ФУНКЦИЯ И МАССИВ ФИГУР
// ==========================================
int main(int argc, const char * argv[]) {
    @autoreleasepool {
        // Создаем объекты разных геометрических фигур
        Rectangle *rect = [[Rectangle alloc] initWithWidth:5.0 height:10.0];
        Circle *circle = [[Circle alloc] initWithRadius:3.0];
        Triangle *triangle = [[Triangle alloc] initWithSideA:3.0 sideB:4.0 sideC:5.0];
        
        // Помещаем их в полиморфный массив NSArray
        NSArray<Figure *> *figures = @[rect, circle, triangle];
        
        NSLog(@"--- Вывод информации о %lu фигурах ---", (unsigned long)figures.count);
        
        // Цикл для перебора всех фигур и вывода информации
        for (Figure *figure in figures) {
            [figure printInfo]; // Вызывается переопределенный метод каждого подкласса
        }
    }
    return 0;
}
