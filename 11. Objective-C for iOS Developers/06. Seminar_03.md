  - Реализация классов (Реализация методов и dealloc)

```
#import "Bird.h"

// --- Bird ---
@implementation Bird
- (instancetype)initWithName:(NSString *)name {
    self = [super init];
    if (self) {
        _name = [name retain]; // Увеличиваем счетчик удержания строки
    }
    return self;
}
- (void)fly { NSLog(@"%@ летит стандартно.", _name); }
- (void)fall { NSLog(@"%@ падает стандартно.", _name); }
- (void)eat { NSLog(@"%@ клюет зерно.", _name); }
- (void)dealloc {
    NSLog(@"[Dealloc] Bird %@ освобожден из памяти.", _name);
    [_name release]; // Освобождаем память переменной
    [super dealloc]; // Обязательный вызов в MRC
}
@end

// --- SuperBird ---
@implementation SuperBird
- (void)fly { NSLog(@"%@ летит со сверхзвуковой скоростью!", _name); }
- (void)fall { NSLog(@"%@ камнем падает вниз, но выживает.", _name); }
- (void)eat { NSLog(@"%@ ест очень быстро.", _name); }
@end

// --- SlowBird ---
@implementation SlowBird
- (void)fly { NSLog(@"%@ лениво машет крыльями.", _name); }
- (void)fall { NSLog(@"%@ плавно планирует вниз.", _name); }
- (void)eat { NSLog(@"%@ долго жует еду.", _name); }
@end

// --- Nest ---
@implementation Nest
@synthesize area = _area;
@synthesize occupancy = _occupancy;

- (void)setBirdRetain:(Bird *)bird {
    if (_bird != bird) {
        [_bird release];
        _bird = [bird retain]; // Сильное удержание -> Цикл удержания
    }
}
- (void)setBirdAssign:(Bird *)bird {
    _bird = bird; // Слабое удержание без подсчета ссылок -> Нет цикла
}
- (Bird *)bird { return _bird; }

- (void)dealloc {
    NSLog(@"[Dealloc] Nest в зоне %@ разрушено.", _area);
    [_bird release]; // Освобождаем, если удерживали сильно
    [_area release];
    [super dealloc];
}
@end
```

  - Воспроизведение и решение Retain Cycle (Задания 3 и 4)   Добавим в класс Bird свойство для гнезда @property (nonatomic, retain) Nest *nest; (с соответствующим синтезом и release в dealloc)

```
void демонстрацияRetainCycle() {
    NSLog(@"--- Старт симуляции Retain Cycle ---");
    
    Bird *superBird = [[SuperBird alloc] initWithName:@"Супер-Орел"];
    Nest *nest = [[Nest alloc] init];
    nest.area = @"Скала";
    
    // Создаем перекрестные сильные ссылки
    superBird.nest = nest;
    [nest setBirdRetain:superBird]; 
    
    // Освобождаем наши локальные указатели
    [superBird release];
    [nest release];
    
    NSLog(@"--- Конец симуляции. Логи dealloc не вывелись! Объекты утекли. ---");
}

void решениеRetainCycle() {
    NSLog(@"--- Старт симуляции БЕЗ Retain Cycle ---");
    
    Bird *slowBird = [[SlowBird alloc] initWithName:@"Ленивый Голубь"];
    Nest *nest = [[Nest alloc] init];
    nest.area = @"Дерево";
    
    slowBird.nest = nest;
    [nest setBirdAssign:slowBird]; // Используем слабую связь assign
    
    [slowBird release]; // Вызовется dealloc для slowBird, который освободит nest
    [nest release];
    
    NSLog(@"--- Конец симуляции. Объекты успешно удалены! ---");
}
```

Улучшение программы:   
Перевод на ARC и АвтоосвобождениеВ режиме ARC (флаг компиляции -fobjc-arc) компилятор сам расставляет вызовы retain и release. Мы убираем их из кода, а связь в Nest делаем по-настоящему weak.

```
// --- Nest.h в ARC ---
@interface Nest : NSObject
@property (nonatomic, weak) Bird *bird; // Автоматическое зануление при уничтожении птицы
@property (nonatomic, strong) NSString *area;
@property (nonatomic) NSInteger occupancy;
@end

// --- Использование @autoreleasepool (Задание 1) ---
int main(int argc, const char * argv[]) {
    @autoreleasepool {
        // Все объекты, созданные здесь внутри циклов или через фабричные методы,
        // будут освобождены сразу при выходе из этого блока памяти.
        
        for (int i = 0; i < 1000; i++) {
            @autoreleasepool { // Локальный пул для оптимизации пикового потребления памяти
                Bird *b = [[SlowBird alloc] initWithName:[NSString stringWithFormat:@"Птица-%d", i]];
                [b fly];
                // b автоматически освобождается здесь
            }
        }
    } // Глобальные объекты освобождаются здесь
    return 0;
}
```


2. Задание 5.    
   - Реализация геттеров и сеттеров вручную в MRC   
   При ручном управлении памятью (MRC) важно правильно обрабатывать старые значения, чтобы избежать утечек памяти или преждевременного удаления объектов.1. @property (nonatomic, retain) id object1;

```
- (id)object1 {
    return _object1;
}
- (void)setObject1:(id)newObject {
    if (_object1 != newObject) {
        [_object1 release];          // Освобождаем старый объект
        _object1 = [newObject retain]; // Удерживаем новый объект
    }
}
```

  - @property (nonatomic, copy) NSMutableArray *array;   
  Примечание: Использование copy на NSMutableArray вернет неизменяемую копию (NSArray). Сеттер должен делать mutableCopy для сохранения изменяемости.

```
- (NSMutableArray *)array {
    return _array;
}
- (void)setArray:(NSMutableArray *)newArray {
    if (_array != newArray) {
        [_array release];
        _array = [newArray mutableCopy]; // Создает новую изменяемую копию с retain count = 1
    }
}
```

  - @property (atomic, retain) NSNumber *number;   
  Потокобезопасность (atomic) реализуется с помощью блокировок (@synchronized).

```
- (NSNumber *)number {
    @synchronized(self) {
        return [[_number retain] autorelease]; // Гарантирует, что объект не умрет в другом потоке при чтении
    }
}
- (void)setNumber:(NSNumber *)newNumber {
    @synchronized(self) {
        if (_number != newNumber) {
            [_number release];
            _number = [newNumber retain];
        }
    }
}
```
