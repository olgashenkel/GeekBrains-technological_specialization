// ************** Задание № 1 **************

#import <Foundation/Foundation.h>
#import <math.h>

int main(int argc, const char * argv[]) {
    @autoreleasepool {
        // ЗАМЕНЯЙТЕ КОНСТАНТЫ ЗДЕСЬ для разных уравнений:
        // Уравнение 1: x^2 - 8x + 12 = 0 -> a = 1, b = -8, c = 12
        // Уравнение 2: 12x^2 - 4x + 2 = 0 -> a = 12, b = -4, c = 2
        // Уравнение 3: x^2 - 100x - 2 = 0 -> a = 1, b = -100, c = -2
        
        double a = 1.0;
        double b = -8.0;
        double c = 12.0;
        
        NSLog(@"Решение уравнения: %.2f*x^2 + (%.2f)*x + (%.2f) = 0", a, b, c);
        
        if (a == 0) {
            if (b == 0) {
                NSLog(@"Ошибка: Уравнение не имеет смысла (0 = 0 или c = 0).");
            } else {
                double x = -c / b;
                NSLog(@"Это линейное уравнение. Корень x = %.4f", x);
            }
        } else {
            double discriminant = b * b - 4 * a * c;
            NSLog(@"Дискриминант D = %.4f", discriminant);
            
            if (discriminant > 0) {
                double x1 = (-b + sqrt(discriminant)) / (2 * a);
                double x2 = (-b - sqrt(discriminant)) / (2 * a);
                NSLog(@"Уравнение имеет два корня:");
                NSLog(@"x1 = %.4f", x1);
                NSLog(@"x2 = %.4f", x2);
            } else if (discriminant == 0) {
                double x = -b / (2 * a);
                NSLog(@"Уравнение имеет один корень:");
                NSLog(@"x = %.4f", x);
            } else {
                // Комплексные корни
                double realPart = -b / (2 * a);
                double imaginaryPart = sqrt(-discriminant) / (2 * a);
                NSLog(@"Уравнение имеет комплексные корни:");
                NSLog(@"x1 = %.4f + %.4fi", realPart, imaginaryPart);
                NSLog(@"x2 = %.4f - %.4fi", realPart, imaginaryPart);
            }
        }
    }
    return 0;
}
