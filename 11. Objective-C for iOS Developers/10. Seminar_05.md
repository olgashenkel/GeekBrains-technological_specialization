2. Дизайн и кастомизация ячейки (CustomTableViewCell)

Ячейка кастомизирована программно (без Storyboard): внутри стандартного contentView создается слой-подложка (cardView), формирующий эффект карточки с тенью и скруглениями.

Класс CustomTableViewCell.h
```
#import <UIKit/UIKit.h>
@interface CustomTableViewCell : UITableViewCell
@property (nonatomic, strong) UIImageView *customImageView;@property (nonatomic, strong) UILabel *customTitleLabel;
@end
```

Класс CustomTableViewCell.m
```
#import "CustomTableViewCell.h"
@implementation CustomTableViewCell {
    UIView *_cardView; // Контейнер для кастомизации дизайна
}

- (instancetype)initWithStyle:(UITableViewCellStyle)style reuseIdentifier:(NSString *)reuseIdentifier {
    self = [super initWithStyle:style reuseIdentifier:reuseIdentifier];
    if (self) {
        self.backgroundColor = [UIColor clearColor];
        self.selectionStyle = UITableViewCellSelectionStyleNone;
        [self setupCardView];
        [self setupSubviews];
    }
    return self;
}

- (void)setupCardView {
    _cardView = [[UIView alloc] init];
    _cardView.backgroundColor = [UIColor whiteColor];
    _cardView.layer.cornerRadius = 14.0;
    
    // Настройка кастомной тени (Эффект глубины)
    _cardView.layer.shadowColor = [UIColor blackColor].CGColor;
    _cardView.layer.shadowOffset = CGSizeMake(0, 4);
    _cardView.layer.shadowRadius = 6.0;
    _cardView.layer.shadowOpacity = 0.08;
    _cardView.layer.masksToBounds = NO;
    
    [self.contentView addSubview:_cardView];
}

- (void)setupSubviews {
    // Кастомное изображение
    self.customImageView = [[UIImageView alloc] init];
    self.customImageView.contentMode = UIViewContentModeScaleAspectFill;
    self.customImageView.layer.cornerRadius = 10.0;
    self.customImageView.clipsToBounds = YES;
    self.customImageView.backgroundColor = [UIColor systemGray6Color];
    [_cardView addSubview:self.customImageView];
    
    // Кастомный текст
    self.customTitleLabel = [[UIImageView alloc] init]; // UILabel
    self.customTitleLabel = [[UILabel alloc] init];
    self.customTitleLabel.numberOfLines = 0; // Разрешаем перенос строк
    self.customTitleLabel.font = [UIFont systemFontOfSize:16 weight:UIFontWeightMedium];
    self.customTitleLabel.textColor = [UIColor darkGrayColor];
    [_cardView addSubview:self.customTitleLabel];
}
// Программная верстка (Auto Layout через фреймы для высокой производительности)
- (void)layoutSubviews {
    [super layoutSubviews];
    
    // Отступы для карточки внутри ячейки
    CGFloat cardPaddingX = 16.0;
    CGFloat cardPaddingY = 8.0;
    _cardView.frame = CGRectMake(cardPaddingX, cardPaddingY, self.contentView.bounds.size.width - (cardPaddingX * 2), self.contentView.bounds.size.height - (cardPaddingY * 2));
    
    // Размеры элементов внутри карточки
    CGFloat elementPadding = 12.0;
    CGFloat imageSize = 70.0;
    
    self.customImageView.frame = CGRectMake(elementPadding, elementPadding, imageSize, imageSize);
    
    CGFloat labelX = imageSize + (elementPadding * 2);
    CGFloat labelWidth = _cardView.bounds.size.width - labelX - elementPadding;
    CGFloat labelHeight = _cardView.bounds.size.height - (elementPadding * 2);
    
    self.customTitleLabel.frame = CGRectMake(labelX, elementPadding, labelWidth, labelHeight);
}
@end
```

3. Сетевой запрос и управление таблицей (ViewController)   
В контроллере реализуется асинхронный GET-запрос. Изображения скачиваются в фоновом потоке прямо во время скролла таблицы и кэшируются в экземплярах модели, чтобы избежать дерганья интерфейса и повторных запросов.


Класс ViewController.m
```
#import "ViewController.h"#import "DataModel.h"#import "CustomTableViewCell.h"
@interface ViewController () <UITableViewDataSource, UITableViewDelegate>
@property (nonatomic, strong) UITableView *tableView;@property (nonatomic, strong) NSMutableArray<DataModel *> *tableData;@property (nonatomic, strong) UIActivityIndicatorView *loadingIndicator;
@end
@implementation ViewController

- (void)viewDidLoad {
    [super viewDidLoad];
    
    self.title = @"Сеть и Кастомизация";
    self.view.backgroundColor = [UIColor systemGray6Color]; // Приятный светлый фон под карточки
    
    self.tableData = [[NSMutableArray alloc] init];
    
    [self setupTableView];
    [self setupLoadingIndicator];
    [self fetchDataFromNetwork];
}

- (void)setupTableView {
    self.tableView = [[UITableView alloc] initWithFrame:self.view.bounds style:UITableViewStylePlain];
    self.tableView.backgroundColor = [UIColor clearColor];
    self.tableView.separatorStyle = UITableViewCellSeparatorStyleNone; // Убираем дефолтные разделители
    self.tableView.dataSource = self;
    self.tableView.delegate = self;
    
    // Регистрация кастомного класса ячейки
    [self.tableView registerClass:[CustomTableViewCell class] forCellReuseIdentifier:@"CustomCellIdentifier"];
    
    [self.view addSubview:self.tableView];
}

- (void)setupLoadingIndicator {
    self.loadingIndicator = [[UIActivityIndicatorView alloc] initWithActivityStyle:UIActivityIndicatorViewStyleLarge];
    self.loadingIndicator.center = self.view.center;
    [self.view addSubview:self.loadingIndicator];
    [self.loadingIndicator startAnimating];
}
#pragma mark - СЕТЕВОЙ GET-ЗАПРОС

- (void)fetchDataFromNetwork {
    NSString *urlString = @"https://typicode.com"; // Ограничимся 50 элементами
    NSURL *url = [NSURL URLWithString:urlString];
    
    NSURLSessionDataTask *task = [[NSURLSession sharedSession] dataTaskWithURL:url completionHandler:^(NSData *data, NSURLResponse *response, NSError *error) {
        
        // Обработка ошибок сети
        if (error) {
            NSLog(@"Ошибка сети: %@", error.localizedDescription);
            return;
        }
        
        NSError *jsonError;
        NSArray *jsonArray = [NSJSONSerialization JSONObjectWithData:data options:kNilOptions error:&jsonError];
        
        if (!jsonError && [jsonArray isKindOfClass:[NSArray class]]) {
            for (NSDictionary *dict in jsonArray) {
                DataModel *model = [[DataModel alloc] initWithDictionary:dict];
                [self.tableData addObject:model];
            }
            
            // Обновляем UI строго в главном потоке
            dispatch_async(dispatch_get_main_queue(), ^{
                [self.loadingIndicator stopAnimating];
                [self.tableView reloadData];
            });
        }
    }];
    
    [task resume]; // Запуск сетевой задачи
}
#pragma mark - UITableView Data Source & Delegate

- (NSInteger)tableView:(UITableView *)tableView numberOfRowsInSection:(NSInteger)section {
    return self.tableData.count;
}

- (CGFloat)tableView:(UITableView *)tableView heightForRowAtIndexPath:(NSIndexPath *)indexPath {
    return 110.0; // Фиксированная высота ячейки с учетом отступов карточки
}

- (UITableViewCell *)tableView:(UITableView *)tableView cellForRowAtIndexPath:(NSIndexPath *)indexPath {
    CustomTableViewCell *cell = [tableView dequeueReusableCellWithIdentifier:@"CustomCellIdentifier" forIndexPath:indexPath];
    
    DataModel *model = self.tableData[indexPath.row];
    cell.customTitleLabel.text = model.titleText;
    
    // АСИНХРОННАЯ ЗАГРУЗКА КАРТИНКИ
    if (model.cachedImage) {
        // Если картинка уже скачана, берем из памяти (мгновенно)
        cell.customImageView.image = model.cachedImage;
    } else {
        // Ставим плейсхолдер на время загрузки
        cell.customImageView.image = [UIImage systemImageNamed:@"photo"]; 
        cell.customImageView.tintColor = [UIColor systemGray4Color];
        
        NSURL *imageUrl = [NSURL URLWithString:model.imageUrlString];
        
        [[[NSURLSession sharedSession] dataTaskWithURL:imageUrl completionHandler:^(NSData *data, NSURLResponse *response, NSError *error) {
            if (data && !error) {
                UIImage *downloadedImage = [UIImage imageWithData:data];
                model.cachedImage = downloadedImage; // Сохраняем в кэш модели
                
                dispatch_async(dispatch_get_main_queue(), ^{
                    // Проверяем видимость ячейки перед обновлением во избежание сбоев при быстром скролле
                    CustomTableViewCell *updateCell = [tableView cellForRowAtIndexPath:indexPath];
                    if (updateCell) {
                        updateCell.customImageView.image = downloadedImage;
                    }
                });
            }
        }] resume];
    }
    
    return cell;
}
@end
```
