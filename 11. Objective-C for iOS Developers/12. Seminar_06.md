1. Хранилища данных в iOS   
В iOS для каждого типа данных подходит свое хранилище.    
Для работы с KeyChain используется встроенный фреймворк Security. Для работы с YapDatabase (актуальной NoSQL-базой данных на основе SQLite) необходимо подключить одноименный Pod (pod 'YapDatabase').

StorageManager.h
```
#import <Foundation/Foundation.h>
@interface StorageManager : NSObject

+ (instancetype)sharedManager;
// 1. UserDefaults (Настройки, легкие данные)
- (void)saveToUserDefaults:(NSString *)value forKey:(NSString *)key;
- (NSString *)readFromUserDefaultsWithKey:(NSString *)key;

// 2. KeyChain (Пароли, токены, приватные данные)
- (BOOL)saveToKeyChain:(NSString *)secret forKey:(NSString *)key;
- (NSString *)readFromKeyChainWithKey:(NSString *)key;

// 3. PropertyList (Структурированные конфиги, словари, массивы)
- (void)saveToPlist:(NSDictionary *)dict fileName:(NSString *)name;
- (NSDictionary *)readFromPlistWithFileName:(NSString *)name;

// 4. FileManager (Бинарные файлы, изображения, крупные документы)
- (void)saveDataToFile:(NSData *)data fileName:(NSString *)name;
- (NSData *)readDataFromFileWithFileName:(NSString *)name;

// 5. YapDatabase (Продвинутая NoSQL база данных)
- (void)saveToYapDatabase:(id)object forKey:(NSString *)key collection:(NSString *)collection;
- (id)readFromYapDatabaseWithKey:(NSString *)key collection:(NSString *)collection;
@end

StorageManager.m
```
#import "StorageManager.h"#import <Security/Security.h>#import <YapDatabase/YapDatabase.h> // Требуется установка через CocoaPods
@implementation StorageManager {
    YapDatabase *_database;
    YapDatabaseConnection *_databaseConnection;
}

+ (instancetype)sharedManager {
    static StorageManager *sharedInstance = nil;
    static dispatch_once_t onceToken;
    dispatch_once(&onceToken, ^{
        sharedInstance = [[self alloc] init];
    });
    return sharedInstance;
}

- (instancetype)init {
    self = [super init];
    if (self) {
        // Инициализация YapDatabase в директории Application Support
        NSString *databasePath = [NSSearchPathForDirectoriesInDomains(NSApplicationSupportDirectory, NSUserDomainMask, YES) firstObject];
        databasePath = [databasePath stringByAppendingPathComponent:@"YapDatabase.sqlite"];
        _database = [[YapDatabase alloc] initWithPath:databasePath];
        _databaseConnection = [_database newConnection];
    }
    return self;
}

// --- 1. USER DEFAULTS ---
- (void)saveToUserDefaults:(NSString *)value forKey:(NSString *)key {
    [[NSUserDefaults standardUserDefaults] setObject:value forKey:key];
    [[NSUserDefaults standardUserDefaults] synchronize];
}

- (NSString *)readFromUserDefaultsWithKey:(NSString *)key {
    return [[NSUserDefaults standardUserDefaults] stringForKey:key];
}

// --- 2. KEYCHAIN ---
- (BOOL)saveToKeyChain:(NSString *)secret forKey:(NSString *)key {
    NSData *data = [secret dataUsingEncoding:NSUTF8StringEncoding];
    NSDictionary *query = @{
        (__bridge id)kSecClass: (__bridge id)kSecClassGenericPassword,
        (__bridge id)kSecAttrAccount: key,
        (__bridge id)kSecValueData: data
    };
    SecItemDelete((__bridge CFDictionaryRef)query); // Удаляем старое, если есть
    OSStatus status = SecItemAdd((__bridge CFDictionaryRef)query, NULL);
    return status == errSecSuccess;
}

- (NSString *)readFromKeyChainWithKey:(NSString *)key {
    NSDictionary *query = @{
        (__bridge id)kSecClass: (__bridge id)kSecClassGenericPassword,
        (__bridge id)kSecAttrAccount: key,
        (__bridge id)kSecReturnData: (__bridge id)kCFBooleanTrue,
        (__bridge id)kSecMatchLimit: (__bridge id)kSecMatchLimitOne
    };
    CFTypeRef dataTypeRef = NULL;
    OSStatus status = SecItemCopyMatching((__bridge CFDictionaryRef)query, &dataTypeRef);
    if (status == errSecSuccess) {
        NSData *data = (__bridge_transfer NSData *)dataTypeRef;
        return [[NSString alloc] initWithData:data encoding:NSUTF8StringEncoding];
    }
    return nil;
}

// --- 3. PROPERTY LIST (PLIST) ---
- (void)saveToPlist:(NSDictionary *)dict fileName:(NSString *)name {
    NSString *path = [NSSearchPathForDirectoriesInDomains(NSDocumentDirectory, NSUserDomainMask, YES) firstObject];
    NSString *finalPath = [path stringByAppendingPathComponent:[NSString stringWithFormat:@"%@.plist", name]];
    [dict writeToFile:finalPath atomically:YES];
}

- (NSDictionary *)readFromPlistWithFileName:(NSString *)name {
    NSString *path = [NSSearchPathForDirectoriesInDomains(NSDocumentDirectory, NSUserDomainMask, YES) firstObject];
    NSString *finalPath = [path stringByAppendingPathComponent:[NSString stringWithFormat:@"%@.plist", name]];
    return [NSDictionary dictionaryWithContentsOfFile:finalPath];
}

// --- 4. FILE MANAGER ---
- (void)saveDataToFile:(NSData *)data fileName:(NSString *)name {
    NSString *path = [NSSearchPathForDirectoriesInDomains(NSCachesDirectory, NSUserDomainMask, YES) firstObject];
    NSString *finalPath = [path stringByAppendingPathComponent:name];
    [[NSFileManager defaultManager] createFileAtPath:finalPath contents:data attributes:nil];
}

- (NSData *)readDataFromFileWithFileName:(NSString *)name {
    NSString *path = [NSSearchPathForDirectoriesInDomains(NSCachesDirectory, NSUserDomainMask, YES) firstObject];
    NSString *finalPath = [path stringByAppendingPathComponent:name];
    return [[NSFileManager defaultManager] contentsAtPath:finalPath];
}

// --- 5. YAPDATABASE ---
- (void)saveToYapDatabase:(id)object forKey:(NSString *)key collection:(NSString *)collection {
    [_databaseConnection readWriteWithBlock:^(YapDatabaseReadWriteTransaction *transaction) {
        [transaction setObject:object forKey:key inCollection:collection];
    }];
}

- (id)readFromYapDatabaseWithKey:(NSString *)key collection:(NSString *)collection {
    __block id object = nil;
    [_databaseConnection readWithBlock:^(YapDatabaseReadTransaction *transaction) {
        object = [transaction objectForKey:key inCollection:collection];
    }];
    return object;
}
@end

2. Безопасная обработка исключений (Предотвращение Crash)  

```
- (void)demonstrateSafeExceptionHandling {
    NSArray *shortArray = @[@"Элемент 0", @"Элемент 1"];
    
    @try {
        NSLog(@"Попытка спровоцировать краш...");
        // Экстренно вызываем обращение по несуществующему индексу (Индекс 5 при длине 2)
        NSString *invalidElement = [shortArray objectAtIndex:5]; 
        NSLog(@"Этот лог никогда не выполнится: %@", invalidElement);
    }
    @catch (NSException *exception) {
        // Перехватываем исключение. Приложение НЕ падает.
        NSLog(@"--- ИСКЛЮЧЕНИЕ ПЕРЕХВАЧЕНО УСПЕШНО ---");
        NSLog(@"Название: %@", exception.name);
        NSLog(@"Причина: %@", exception.reason);
        NSLog(@"Стек вызовов: %@", exception.callStackSymbols);
    }
    @finally {
        // Блок выполняется всегда, независимо от того, был краш или нет
        NSLog(@"Выполнение блока @finally для очистки ресурсов.");
    }
    
    NSLog(@"Приложение продолжает работу в штатном режиме!");
}

3. Objective-C Runtime и Method Swizzling   
Method Swizzling — это подмена имплементации метода прямо во время выполнения программы (Runtime). С его помощью можно переопределить методы жизненного цикла или внедриться во внутренние скрытые методы Apple (Private SDK).   


UIViewController+Swizzling.m
```
#import <UIKit/UIKit.h>#import <objc/runtime.h>
@implementation UIViewController (Swizzling)
// Метод +load вызывается автоматически при загрузке класса в память Runtime
+ (void)load {
    static dispatch_once_t onceToken;
    dispatch_once(&onceToken, ^{
        Class class = [self class];

        // 1. Подмена метода viewDidLoad
        [self swizzleInstanceMethodOfClass:class 
                          originalSelector:@selector(viewDidLoad) 
                           swizzledSelector:@selector(custom_viewDidLoad)];

        // 2. Подмена метода viewWillAppear:
        [self swizzleInstanceMethodOfClass:class 
                          originalSelector:@selector(viewWillAppear:) 
                           swizzledSelector:@selector(custom_viewWillAppear:)];
        
        // 3. Подмена приватного метода Apple из Private SDK (_handleLongTapEvent)
        // Внимание: Селектор создается из строки, так как метода нет в публичных хедерах.
        SEL privateSelector = NSSelectorFromString(@"_handleLongTapEvent:");
        if ([self instancesRespondToSelector:privateSelector]) {
            [self swizzleInstanceMethodOfClass:class 
                              originalSelector:privateSelector 
                               swizzledSelector:@selector(custom_handleLongTapEvent:)];
        }
    });
}

+ (void)swizzleInstanceMethodOfClass:(Class)class originalSelector:(SEL)originalSelector swizzledSelector:(SEL)swizzledSelector {
    Method originalMethod = class_getInstanceMethod(class, originalSelector);
    Method swizzledMethod = class_getInstanceMethod(class, swizzledSelector);

    // Пытаемся добавить метод, если его нет в текущем классе (например, он у родителя)
    BOOL didAddMethod = class_addMethod(class,
                                        originalSelector,
                                        method_getImplementation(swizzledMethod),
                                        method_getTypeEncoding(swizzledMethod));

    if (didAddMethod) {
        class_replaceMethod(class,
                            swizzledSelector,
                            method_getImplementation(originalMethod),
                            method_getTypeEncoding(originalMethod));
    } else {
        // Если метод уже есть, просто меняем имплементации местами
        method_exchangeImplementations(originalMethod, swizzledMethod);
    }
}
#pragma mark - Кастомные имплементации (Кастомный код)

- (void)custom_viewDidLoad {
    // Внутри подмененного метода вызов самого себя "custom_viewDidLoad" 
    // на самом деле вызывает ОРИГИНАЛЬНЫЙ viewDidLoad из-за обмена имплементациями.
    [self custom_viewDidLoad]; 
    
    NSLog(@"[Runtime Swizzling] Вызван viewDidLoad для экрана: %@", NSStringFromClass([self class]));
}

- (void)custom_viewWillAppear:(BOOL)animated {
    [self custom_viewWillAppear:animated]; // Вызов оригинального метода
    
    NSLog(@"[Runtime Swizzling] Вызван viewWillAppear для экрана: %@", NSStringFromClass([self class]));
}
// Кастомный перехватчик приватного метода Apple
- (void)custom_handleLongTapEvent:(id)event {
    NSLog(@"[Runtime Private SDK] Перехвачено долгое нажатие (_handleLongTapEvent:) на экране: %@", NSStringFromClass([self class]));
    
    // Передаем управление дальше оригинальному методу Apple, чтобы не сломать жесты
    if ([self respondsToSelector:@selector(custom_handleLongTapEvent:)]) {
        [self custom_handleLongTapEvent:event];
    }
}
@end
